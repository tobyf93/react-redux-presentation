	this.age = 0;

function Person() {
	this.age = 0;

	setInterval((function() {
		this.age++;
	}).bind(this), 1000);
}

var p = new Person();
  					  
							
function Person() {
	var that = this;
	this.age = 0;

	setInterval(function() {
		that.age++;
	}, 1000);
}

var p = new Person();

	this.age = 0;
	setInterval(() => this.age++, 1000);
}

var p = new Person();
